
# THIS = Province, FROM = Country, ROOT = Merc Unit
on_mercenary_recruited = {
    if = {
        limit = {
            LAFI_is_free_merc_company = yes
            FROM = { has_not_free_mercs_trait = yes }
        }
        if = {
            limit = {
                FROM = {
                    OR = {
                        has_country_flag = fifty_percent_merc_army_professionalism_cost
                        has_country_modifier = pol_cheap_ap_mercs_modifier
                    }
                }
            }
            FROM = { add_army_professionalism = 0.025 }
        }
        else = {
            FROM = { add_army_professionalism = 0.05 }
        }
    }
}

# root = winning country, from = loser country
on_main_war_won = {
    if = {
        limit = { has_ruler_flag = LAFI_WuFuWang_ruler_flag }
        change_variable = {
            which = lafi_wufuwang_war_won_count_var
            value = 1
        }
    }
    if = {
        limit = {
            has_ruler_flag = LAFI_WuFuWang_ruler_flag
            check_variable = {
                which = lafi_wufuwang_war_won_count_var
                value = 10
            }
            not = {
                has_ruler_flag = LAFI_WuFuWang_personality_added_1_flag
            }
            adm = 3
            dip = 3
            mil = 3
        }
        add_ruler_personality = legendary_conqueror_personality
        set_ruler_flag = LAFI_WuFuWang_personality_added_1_flag
    }
    if = {
        limit = {
            has_ruler_flag = LAFI_WuFuWang_ruler_flag
            check_variable = {
                which = lafi_wufuwang_war_won_count_var
                value = 25
            }
            not = {
                has_ruler_flag = LAFI_WuFuWang_personality_added_2_flag
            }
            adm = 3
            dip = 3
            mil = 3
        }
        add_ruler_personality = iron_crowned_personality
        set_ruler_flag = LAFI_WuFuWang_personality_added_2_flag
    }
    if = {
        limit = {
            has_ruler_flag = LAFI_WuFuWang_ruler_flag
            check_variable = {
                which = lafi_wufuwang_war_won_count_var
                value = 50
            }
            not = {
                has_ruler_flag = LAFI_flavor_events_2_triggered_flag
            }
            adm = 6
            dip = 6
            mil = 6
        }
        country_event = {
            id = LAFI_flavor_events.2
        }
    }
}

# root = loser country, from = winner country
on_war_lost = {
    ROOT = {
        if = {
            limit = {
                has_country_flag = LAFI_mil_ideas_4_japanese_primary_modifier
            }
            remove_country_flag = LAFI_mil_ideas_4_japanese_primary_modifier
        }
    }
}

on_main_war_lost = {
    ROOT = {
        if = {
            limit = {
                has_ruler_flag = LAFI_WuFuWang_ruler_flag
            }
            if = {
                limit = { adm = 1 }
                change_adm = -1
            }
            if = {
                limit = { dip = 1 }
                change_dip = -1
            }
            if = {
                limit = { mil = 1 }
                change_mil = -1
            }
            if = {
                limit = {
                    or = {
                        adm = 1
                        dip = 1
                        mil = 1
                    }
                }
                every_country = {
                    add_aggressive_expansion = {
                        who = ROOT
                        value = -20
                    }
                }
            }
        }
    }
}

on_monthly_pulse = {
    if = {
        limit = {
            has_country_flag = LAFI_recruit_Liewartine_decision_1_a_decided_flag
        }
        random_owned_province = {
            limit = {
                or = {
                    trade_goods = grain
                    trade_goods = livestock
                }
                LAFI_province_can_exploit_tax_scripted_trigger = yes
            }
            LAFI_exploit_all_tax_scripted_effect = yes
            set_base_tax = 1
        }
        ROOT = {
            clr_country_flag = LAFI_recruit_Liewartine_decision_1_a_decided_flag
        }

    }
}
